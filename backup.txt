import datetime
import os
import pickle

import models

today = datetime.datetime.today()
Posts = models.BlogPost.all().fetch(1000)
fname = os.path.join(os.environ['HOME'],
   today.strftime('Dropbox/The_Grand_Locus/posts_%Y%m%d.bkp'))
pickle.dump(Posts, open(fname, 'w'))
